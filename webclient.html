Service

@autoinject
export class CompletedJobsService {
    constructor(private httpClient: HttpClient){
    }

	loadWorkitemTypes() : Promise {
        return this.httpClient.fetch('/batch/workitemTypes')
            .then(response => response.json());
    }
}

Service Aufruf

this.completedJobsService.loadWorkitemTypes()
    .then(data => {
	for( let entry of data ){
		this.workitemTypesArray
		.push(new IdNamePair(entry.Key, entry.Value));
	}
});
